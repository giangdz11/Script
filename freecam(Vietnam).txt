local RunService = game:GetService("RunService")
local UserInputService = game:GetService("UserInputService")
local Players = game:GetService("Players")

local cam = workspace.CurrentCamera
local player = Players.LocalPlayer
local active = false
local editMode = false
local cameraLocked = false 
local showLockButton = true 
local baseSpeed = 0.5 -- Tốc độ mặc định ban đầu khá chậm
local moveDir = Vector3.new(0,0,0)
local cameraRotation = Vector2.new(0, 0)
local sensitivity = 0.5
local isInteractingWithUI = false

local savedPositions = {}

-- --- GIAO DIỆN CHÍNH ---
local screenGui = Instance.new("ScreenGui", player.PlayerGui)
screenGui.Name = "FreecamPrecisionGui"
screenGui.ResetOnSpawn = false
screenGui.ZIndexBehavior = Enum.ZIndexBehavior.Sibling

-- Nút Menu ⚙️
local menuBtn = Instance.new("TextButton", screenGui)
menuBtn.Size = UDim2.new(0, 45, 0, 45)
menuBtn.Position = UDim2.new(0, 15, 0.05, 0)
menuBtn.Text = "⚙️"
menuBtn.BackgroundColor3 = Color3.fromRGB(45, 45, 45)
menuBtn.TextColor3 = Color3.new(1,1,1)
menuBtn.ZIndex = 100
Instance.new("UICorner", menuBtn).CornerRadius = UDim.new(0, 8)

-- Bảng tùy chỉnh
local settingsFrame = Instance.new("Frame", screenGui)
settingsFrame.Size = UDim2.new(0, 220, 0, 350)
settingsFrame.Position = UDim2.new(0.5, -110, 0.15, 0)
settingsFrame.BackgroundColor3 = Color3.fromRGB(30, 30, 30)
settingsFrame.Visible = false
settingsFrame.ZIndex = 101

local layout = Instance.new("UIListLayout", settingsFrame)
layout.Padding = UDim.new(0, 8)
layout.HorizontalAlignment = Enum.HorizontalAlignment.Center
Instance.new("UIPadding", settingsFrame).PaddingTop = UDim.new(0, 10)

local function createMenuBtn(txt, col)
	local b = Instance.new("TextButton", settingsFrame)
	b.Size = UDim2.new(0, 200, 0, 35)
	b.Text = txt
	b.BackgroundColor3 = col
	b.TextColor3 = Color3.new(1,1,1)
	b.Font = Enum.Font.SourceSansBold
	Instance.new("UICorner", b).CornerRadius = UDim.new(0, 6)
	return b
end

local btnEdit = createMenuBtn("Sửa Vị Trí: TẮT", Color3.fromRGB(180, 100, 0))

-- Khung chỉnh tốc độ (Chế độ chính xác cao)
local speedFrame = Instance.new("Frame", settingsFrame)
speedFrame.Size = UDim2.new(0, 200, 0, 40)
speedFrame.BackgroundTransparency = 1

local btnSpeedDown = Instance.new("TextButton", speedFrame)
btnSpeedDown.Size = UDim2.new(0, 40, 0, 40)
btnSpeedDown.Text = "-"
btnSpeedDown.BackgroundColor3 = Color3.fromRGB(60, 60, 60)
btnSpeedDown.TextColor3 = Color3.new(1,1,1)

local speedLabel = Instance.new("TextLabel", speedFrame)
speedLabel.Size = UDim2.new(0, 120, 0, 40)
speedLabel.Position = UDim2.new(0, 40, 0, 0)
speedLabel.Text = string.format("Tốc độ: %.1f", baseSpeed)
speedLabel.BackgroundColor3 = Color3.fromRGB(45, 45, 45)
speedLabel.TextColor3 = Color3.new(1,1,1)

local btnSpeedUp = Instance.new("TextButton", speedFrame)
btnSpeedUp.Size = UDim2.new(0, 40, 0, 40)
btnSpeedUp.Position = UDim2.new(0, 160, 0, 0)
btnSpeedUp.Text = "+"
btnSpeedUp.BackgroundColor3 = Color3.fromRGB(60, 60, 60)
btnSpeedUp.TextColor3 = Color3.new(1,1,1)

local btnToggleLockUI = createMenuBtn("Hiện Nút Khóa: BẬT", Color3.fromRGB(0, 100, 150))
local btnSave = createMenuBtn("Lưu Vị Trí", Color3.fromRGB(0, 120, 0))
local btnLoad = createMenuBtn("Tải Vị Trí", Color3.fromRGB(0, 100, 180))
local btnReset = createMenuBtn("Đặt Lại Mặc Định", Color3.fromRGB(150, 0, 0))
local btnClose = createMenuBtn("Đóng Menu", Color3.fromRGB(70, 70, 70))

-- Logic Tốc độ (Chỉnh cực chậm)
btnSpeedUp.MouseButton1Click:Connect(function()
	baseSpeed = math.min(baseSpeed + 0.1, 10)
	speedLabel.Text = string.format("Tốc độ: %.1f", baseSpeed)
end)
btnSpeedDown.MouseButton1Click:Connect(function()
	baseSpeed = math.max(baseSpeed - 0.1, 0.1) -- Tối thiểu là 0.1
	speedLabel.Text = string.format("Tốc độ: %.1f", baseSpeed)
end)

-- --- HỆ THỐNG NÚT ĐIỀU KHIỂN ---
local controlButtons = {}
local defaultPos = {
	["Kích Hoạt"] = UDim2.new(0.5, -35, 0.05, 0),
	["Khóa Cam"] = UDim2.new(0.5, -35, 0.17, 0),
	["Tiến"] = UDim2.new(0.8, 0, 0.55, 0),
	["Lùi"] = UDim2.new(0.8, 0, 0.72, 0),
	["Lên"] = UDim2.new(0.1, 0, 0.55, 0),
	["Xuống"] = UDim2.new(0.1, 0, 0.72, 0)
}

local draggingBtn = nil
local dragOffset = Vector2.new(0,0)

local function createControlBtn(name)
	local btn = Instance.new("TextButton", screenGui)
	btn.Name = name
	btn.Size = UDim2.new(0, 70, 0, 70)
	btn.Position = defaultPos[name]
	btn.Text = name
	btn.BackgroundColor3 = Color3.fromRGB(0, 0, 0)
	btn.BackgroundTransparency = 0.4
	btn.TextColor3 = Color3.new(1,1,1)
	btn.Font = Enum.Font.SourceSansBold
	btn.ZIndex = 50
	Instance.new("UICorner", btn).CornerRadius = UDim.new(0, 15)
	
	btn.InputBegan:Connect(function(input)
		if input.UserInputType == Enum.UserInputType.MouseButton1 or input.UserInputType == Enum.UserInputType.Touch then
			if editMode then
				draggingBtn = btn
				local absPos = btn.AbsolutePosition
				dragOffset = Vector2.new(input.Position.X - absPos.X, input.Position.Y - absPos.Y)
			else
				if name == "Kích Hoạt" then
					active = not active
					cam.CameraType = active and Enum.CameraType.Scriptable or Enum.CameraType.Custom
					btn.BackgroundColor3 = active and Color3.fromRGB(0, 180, 0) or Color3.fromRGB(0,0,0)
				elseif name == "Khóa Cam" then
					cameraLocked = not cameraLocked
					btn.Text = cameraLocked and "ĐÃ KHÓA" or "Khóa Cam"
					btn.BackgroundColor3 = cameraLocked and Color3.fromRGB(200, 0, 0) or Color3.fromRGB(0,0,0)
					if cameraLocked then moveDir = Vector3.new(0,0,0) end
				elseif not cameraLocked then
					if name == "Tiến" then moveDir = Vector3.new(0, 0, -1)
					elseif name == "Lùi" then moveDir = Vector3.new(0, 0, 1)
					elseif name == "Lên" then moveDir = Vector3.new(0, 1, 0)
					elseif name == "Xuống" then moveDir = Vector3.new(0, -1, 0) end
				end
			end
		end
	end)

	btn.InputEnded:Connect(function(input)
		if input.UserInputType == Enum.UserInputType.MouseButton1 or input.UserInputType == Enum.UserInputType.Touch then
			if not editMode then
				if name ~= "Kích Hoạt" and name ~= "Khóa Cam" then moveDir = Vector3.new(0, 0, 0) end
			end
			draggingBtn = nil
		end
	end)

	controlButtons[name] = btn
end

for name, _ in pairs(defaultPos) do createControlBtn(name) end

-- Menu Logic
btnToggleLockUI.MouseButton1Click:Connect(function()
	showLockButton = not showLockButton
	controlButtons["Khóa Cam"].Visible = showLockButton
	btnToggleLockUI.Text = showLockButton and "Hiện Nút Khóa: BẬT" or "Hiện Nút Khóa: TẮT"
end)

UserInputService.InputChanged:Connect(function(input)
	if draggingBtn and (input.UserInputType == Enum.UserInputType.MouseMovement or input.UserInputType == Enum.UserInputType.Touch) then
		draggingBtn.Position = UDim2.new(0, input.Position.X - dragOffset.X, 0, input.Position.Y - dragOffset.Y)
	end
end)

btnEdit.MouseButton1Click:Connect(function()
	editMode = not editMode
	btnEdit.Text = editMode and "Sửa Vị Trí: BẬT" or "Sửa Vị Trí: TẮT"
	btnEdit.BackgroundColor3 = editMode and Color3.fromRGB(0, 200, 0) or Color3.fromRGB(180, 100, 0)
end)

btnSave.MouseButton1Click:Connect(function()
	for name, btn in pairs(controlButtons) do savedPositions[name] = btn.Position end
	btnSave.Text = "ĐÃ LƯU!"
	task.wait(0.5)
	btnSave.Text = "Lưu Vị Trí"
end)

btnLoad.MouseButton1Click:Connect(function()
	for name, pos in pairs(savedPositions) do if controlButtons[name] then controlButtons[name].Position = pos end end
end)

btnReset.MouseButton1Click:Connect(function()
	for name, pos in pairs(defaultPos) do controlButtons[name].Position = pos end
	baseSpeed = 0.5
	speedLabel.Text = string.format("Tốc độ: %.1f", baseSpeed)
end)

menuBtn.MouseButton1Click:Connect(function() 
	settingsFrame.Visible = not settingsFrame.Visible 
	isInteractingWithUI = settingsFrame.Visible
end)
btnClose.MouseButton1Click:Connect(function() 
	settingsFrame.Visible = false 
	isInteractingWithUI = false
end)

-- --- VÒNG LẶP CAMERA ---
RunService.RenderStepped:Connect(function()
	if active then
		cam.CameraType = Enum.CameraType.Scriptable
		local rotationCFrame = CFrame.Angles(0, math.rad(cameraRotation.X), 0) * CFrame.Angles(math.rad(cameraRotation.Y), 0, 0)
		cam.CFrame = CFrame.new(cam.CFrame.Position) * rotationCFrame * CFrame.new(moveDir * baseSpeed)
	end
end)

UserInputService.InputChanged:Connect(function(input)
	if active and not editMode and not isInteractingWithUI and not cameraLocked then
		if input.UserInputType == Enum.UserInputType.Touch or input.UserInputType == Enum.UserInputType.MouseMovement then
			cameraRotation = cameraRotation - Vector2.new(input.Delta.X, input.Delta.Y) * sensitivity
		end
	end
end)
